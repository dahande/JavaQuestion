Javaのメソッド定義例

メソッドの定義

Javaでメソッドを定義する際は、メソッド名、引数リスト、戻り値の型、メソッド本体を指定します。

// メソッドの定義
public class MyClass {

    // メソッドの定義
    public void printMessage() {
        System.out.println("Hello, World!"); // メソッドの中で行う処理
    }
    
    // 引数を受け取るメソッドの例
    public void printWithPrefix(String prefix, String message) {
        System.out.println(prefix + ": " + message);
    }
    
    // 戻り値を持つメソッドの例
    public int add(int a, int b) {
        return a + b;
    }
    
    // メソッドをオーバーロードする例
    public double add(double a, double b) {
        return a + b;
    }
}

// メインクラス
public class Main {
    public static void main(String[] args) {
        MyClass myClass = new MyClass();
        myClass.printMessage(); // 出力: Hello, World!
        
        myClass.printWithPrefix("Prefix", "Message"); // 出力: Prefix: Message
        
        int sum = myClass.add(3, 5); // sumは8
        System.out.println("Sum: " + sum);
        
        double sumDouble = myClass.add(3.5, 2.5); // sumDoubleは6.0
        System.out.println("Sum Double: " + sumDouble);
    }
}
